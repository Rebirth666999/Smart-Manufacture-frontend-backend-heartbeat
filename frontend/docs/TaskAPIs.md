- `1`: 新建
- `2`: 审核中

- `4`: 生产中

### 生产计划状态 (mpStat)
- `1`: 新建
- `2`: 审核中
- `3`: 已审核
- `4`: 执行中

- `6`: 已取消

- `1`: 新建
- `2`: 待分配

- `4`: 执行中
- `5`: 已完成
- `6`: 已取消

## 使用示例

```javascript
// 获取所有订单需求
const getOrderDemands = async () => {
  try {
    const response = await axios.get('/system/orderDemand/list');
    return response.data.rows;
  } catch (error) {
    console.error('获取订单需求失败:', error);
    return [];
  }
};

// 获取特定生产计划下的任务
const getTasksByPlan = async (mpCode) => {
  try {
    const response = await axios.get('/system/manufactureTask/list', {
      params: {
        mpCode: mpCode
      }
    });
    return response.data.rows;
  } catch (error) {
    console.error('获取生产任务失败:', error);
    return [];
  }
};
```
